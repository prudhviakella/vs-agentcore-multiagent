# requirements_local.txt

# Local development dependencies — covers all 6 agents + platform.
#
# PYTHON VERSION REQUIREMENT: 3.11+
# ----------------------------------
# This project requires Python 3.11 or higher.
# Python 3.8 (macOS system default) will NOT work.
# langchain>=0.3.0, langgraph, bedrock-agentcore all require Python>=3.9+.
# langgraph-checkpoint-postgres requires Python>=3.10+.
#
# Mac setup (one-time):
#   brew install python@3.11
#   python3.11 -m pip install -r requirements_local.txt
#
# Verify your Python version first:
#   python3.11 --version   # should print Python 3.11.x
#   which python3.11       # should NOT be /usr/bin/python3 (macOS system)
#
# If python3.11 is not found:
#   brew install python@3.11
#   echo 'export PATH="/opt/homebrew/opt/python@3.11/bin:$PATH"' >> ~/.zshrc
#   source ~/.zshrc
#
# Individual agent requirements.txt files are for Docker builds only
# (they install into the container). Do not use them for local dev.
